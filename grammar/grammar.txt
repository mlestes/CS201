unary : NUMBER
      | MINUS NUMBER
      | VARIABLE
      | MINUS VARIABLE
      | OPEN_PAR expression CLOSE_PAR
expression : unary operator unary
           | STRING
           | EMPTY
operator : PLUS
         | TIMES
         | MINUS
         | DIVIDE
         | MOD
         | POW
         | EQUALS
if : IF OPEN_PAR expression CLOSE_PAR block else
else : ELSE block
     | EMPTY
while : WHILE OPEN_PAR expression CLOSE_PAR block
block : BEGIN statementlist END
statementlist : statement
              | statement statementlist
              | EMPTY
statement : expression SEMI_COLON
          | if
          | while
vardef : VAR VARIABLE ASSIGN expression SEMI_COLON
       | VAR VARIABLE SEMI_COLON
funcdef : FUNCTION VARIABLE OPEN_PAR parameterlist CLOSE_PAR block
parameterlist : parameter
              | parameter parameterlist
              | EMPTY
parameter : VARIABLE
funccall : FUNCTION OPEN_PAR parameterlist CLOSE_PAR
comment : OPEN_ANGLE STRING CLOSE_ANGLE
